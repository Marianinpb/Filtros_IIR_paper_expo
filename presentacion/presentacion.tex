\section{Contexto y Motivación}

\begin{frame}{El Problema del Diseño de Filtros IIR}
  \begin{itemize}
    \item Los métodos de transformación (e.g., bilineal) introducen distorsiones (warping) en altas frecuencias.
    \item La \textbf{optimización numérica directa} de los coeficientes es preferida, pero propensa a óptimos locales.
    \item Métodos bioinspirados como \textbf{GSO} (Group Search Optimization) dependen de un único ``productor'', limitando la exploración global.
  \end{itemize}
  \begin{alertblock}{Propuesta: R-GSO}
    Integrar el \textbf{Método de Taguchi} en GSO para seleccionar un productor óptimo a partir de tres candidatos, garantizando robustez matemática y escape de óptimos locales.
  \end{alertblock}
\end{frame}

\section{Formulación Matemática}

\begin{frame}{Función de Transferencia y Costo}
  Filtro IIR en cascada:
  \begin{equation}
    H(\omega, x) = \alpha \gamma \left( \prod_{i=1}^M \frac{1 + a_i e^{-j\omega}}{1 + b_i e^{-j\omega}} \right) \left( \prod_{k=1}^N \frac{1 + c_{1k} e^{-j\omega} + c_{2k} e^{-2j\omega}}{1 + d_{1k} e^{-j\omega} + d_{2k} e^{-2j\omega}} \right)
  \end{equation}
  
  Minimización de función de costo $f(x)$ que unifica 4 criterios:
  \begin{equation}
    \min_x f(x) = \min_x \left[ v_1 E^1_M(x) + v_2 E^2_M(x) + v_3 \delta_1(x) + v_4 \delta_2(x) \right]
  \end{equation}
  Donde $E^1_M, E^2_M$ son errores de norma $L_1, L_2$, y $\delta_1, \delta_2$ son los ripples de banda de paso y rechazo.
\end{frame}

\begin{frame}{Restricciones de Estabilidad}
  Para garantizar que los polos permanezcan dentro del círculo unitario (plano $Z$), se imponen las siguientes inecuaciones:
  \begin{equation*}
    1 + b_i \ge 0, \quad 1 - b_i \ge 0
  \end{equation*}
  \begin{equation*}
    1 - d_{2k} \ge 0, \quad 1 + d_{1k} + d_{2k} \ge 0, \quad 1 - d_{1k} + d_{2k} \ge 0
  \end{equation*}
  Estas restricciones definen un espacio de búsqueda altamente acotado y complejo.
\end{frame}

\section{Algoritmo R-GSO (Integración de Taguchi)}

\begin{frame}{Selección Robusta del Productor}
  En cada iteración, se seleccionan 3 productores: el mejor ($X_{p1}$), el segundo mejor ($X_{p2}$) y uno aleatorio ($X_{p3}$).
  \vspace{0.5cm}
  
  Se aplican \textbf{arreglos ortogonales de Taguchi} para cruzar sistemáticamente estos perfiles. El desempeño se evalúa mediante la relación Señal/Ruido (Robust Response Value):
  \begin{equation}
    E_{fj} = -10 \cdot \log \left( \frac{J_{i(avg)}^2}{J_{Mavg}^2} + \frac{J_{i(std)}^2}{J_{Mstd}^2} \right), \quad j=1, 2
  \end{equation}
  
  Este criterio prioriza no solo un bajo costo promedio, sino una \textbf{baja desviación estándar}, asegurando estabilidad estructural.
\end{frame}

\section{Resultados y Conclusiones}

\begin{frame}{Resultados Experimentales}
  Se evaluaron filtros LP, HP, BP y BS, comparando R-GSO contra GSO estándar y HTGA.
  \begin{itemize}
    \item \textbf{Fitness Final:} R-GSO minimizó de forma superior la función de costo en todos los casos.
    \item \textbf{Convergencia:} Escape eficiente de óptimos locales (especialmente notorio en el filtro Pasa-altas HP).
    \item \textbf{Eficiencia:} Menor cantidad de evaluaciones de la función objetivo respecto a HTGA (72,500 vs 130,000 en LP/HP) gracias al uso estadístico de Taguchi.
  \end{itemize}
\end{frame}

\begin{frame}{Conclusiones}
  \begin{itemize}
    \item La integración formal de la estadística de Taguchi con metaheurísticas de enjambre resuelve la debilidad estructural del GSO clásico.
    \item El algoritmo R-GSO demuestra un rigor matemático superior que se traduce en filtros IIR digitales más exactos y un costo computacional acotado.
  \end{itemize}
\end{frame}
