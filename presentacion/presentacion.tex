\section{Contexto y Motivación}

\begin{frame}{El Problema del Diseño de Filtros IIR}
  \begin{itemize}
    \item Los métodos de transformación (e.g., bilineal) introducen distorsiones (warping) en altas frecuencias.
    \item La \textbf{optimización numérica directa} de los coeficientes es preferida, pero altamente restrictiva y propensa a óptimos locales.
  \end{itemize}
  \begin{alertblock}{Propuesta: R-GSO}
    Integrar el \textbf{Método de Taguchi} en Group Search Optimization (GSO) para seleccionar un productor óptimo a partir de tres candidatos, garantizando robustez matemática y un eficiente escape de óptimos locales.
  \end{alertblock}
\end{frame}

\begin{frame}{Algoritmo GSO: Metáfora de Forrajeo y Limitaciones}
  El GSO se inspira en el comportamiento de forrajeo animal, asignando roles a las soluciones (vectores):
  \begin{itemize}
    \item \textbf{Productor:} Solución con el mejor \textit{fitness} (menor costo). Explora localmente a su alrededor.
    \item \textbf{Scroungers} (Oportunistas): Copian el comportamiento del productor y se dirigen hacia él.
    \item \textbf{Rangers} (Exploradores): Se dispersan de forma aleatoria para mantener la diversidad.
  \end{itemize}
  \vspace{0.3cm}
  \textbf{Problema del GSO Clásico:}
  \begin{itemize}
    \item Al depender de un \textbf{único productor}, si este queda atrapado en un mínimo local, todos los scroungers convergen allí, estancando prematuramente la búsqueda.
  \end{itemize}
\end{frame}

\section{Formulación Matemática}

\begin{frame}{Función de Transferencia y Costo}
  Filtro IIR en cascada:
  \begin{equation}
    H(\omega, x) = \alpha \gamma \left( \prod_{i=1}^M \frac{1 + a_i e^{-j\omega}}{1 + b_i e^{-j\omega}} \right) \left( \prod_{k=1}^N \frac{1 + c_{1k} e^{-j\omega} + c_{2k} e^{-2j\omega}}{1 + d_{1k} e^{-j\omega} + d_{2k} e^{-2j\omega}} \right)
  \end{equation}
  
  Minimización de función de costo $f(x)$ que unifica 4 criterios:
  \begin{equation}
    \min_x f(x) = \min_x \left[ v_1 E^1_M(x) + v_2 E^2_M(x) + v_3 \delta_1(x) + v_4 \delta_2(x) \right]
  \end{equation}
  Donde $E^1_M, E^2_M$ son errores de norma $L_1, L_2$, y $\delta_1, \delta_2$ son los ripples de banda de paso y rechazo.
\end{frame}

\begin{frame}{Restricciones de Estabilidad}
  Para garantizar que los polos permanezcan dentro del círculo unitario (plano $Z$), se imponen las siguientes inecuaciones:
  \begin{equation*}
    1 + b_i \ge 0, \quad 1 - b_i \ge 0
  \end{equation*}
  \begin{equation*}
    1 - d_{2k} \ge 0, \quad 1 + d_{1k} + d_{2k} \ge 0, \quad 1 - d_{1k} + d_{2k} \ge 0
  \end{equation*}
  Estas restricciones definen un espacio de búsqueda altamente acotado y complejo.
\end{frame}

\section{Algoritmo R-GSO (Integración de Taguchi)}

\begin{frame}{Método de Taguchi: Diseño Robusto}
  Busca obtener configuraciones paramétricas con buen desempeño medio y \textbf{baja variabilidad}.
  \begin{itemize}
    \item \textbf{Factores y Niveles:} Los coeficientes del filtro actúan como factores. Sus niveles son tomados de los valores de dos vectores candidatos.
    \item \textbf{Arreglos Ortogonales (ej. $L_8$):} Reducen el espacio combinatorio, balanceando las pruebas para estimar el efecto aislado de cada factor.
    \item \textbf{Métrica de Robustez (S/N):} Se evalúa la media y la desviación estándar penalizando la inestabilidad.
  \end{itemize}
  \begin{equation}
    E_{fj} = -10 \cdot \log \left( \frac{J_{i(avg)}^2}{J_{Mavg}^2} + \frac{J_{i(std)}^2}{J_{Mstd}^2} \right)
  \end{equation}
\end{frame}

\begin{frame}{Integración de Taguchi en el Ciclo GSO}
  El algoritmo R-GSO se ejecuta iterativamente:
  \begin{enumerate}
    \item \textbf{Evaluación:} Medir el costo de toda la población (solo vectores estables).
    \item \textbf{Selección:} Elegir a los tres mejores referentes: el óptimo ($X_{p1}$), el sub-óptimo ($X_{p2}$) y un explorador aleatorio ($X_{p3}$).
    \item \textbf{Cruce Robusto:} Aplicar Taguchi usando $X_{p1}$ y $X_{p2}$ como niveles. Utilizar $X_{p3}$ para inducir variabilidad durante la evaluación de $E_{fj}$.
    \item \textbf{Actualización:} Elegir los niveles que maximizan $E_{fj}$ para conformar un \textbf{nuevo productor óptimo}.
    \item \textbf{Desplazamiento:} Los Scroungers convergen hacia el nuevo productor, mitigando la retención en mínimos locales.
  \end{enumerate}
\end{frame}

\section{Resultados y Conclusiones}

\begin{frame}{Resultados Experimentales}
  Se evaluaron filtros LP, HP, BP y BS, comparando R-GSO contra GSO estándar y HTGA.
  \begin{itemize}
    \item \textbf{Fitness Final:} R-GSO minimizó de forma superior la función de costo en todos los casos.
    \item \textbf{Convergencia:} Escape eficiente de óptimos locales (especialmente notorio en el filtro Pasa-altas HP).
    \item \textbf{Eficiencia:} Menor cantidad de evaluaciones de la función objetivo respecto a HTGA (72,500 vs 130,000 en LP/HP) gracias al uso estadístico de Taguchi.
  \end{itemize}
\end{frame}

\begin{frame}{Conclusiones}
  \begin{itemize}
    \item La integración formal de la estadística de Taguchi con metaheurísticas de enjambre resuelve la debilidad estructural del GSO clásico.
    \item El algoritmo R-GSO demuestra un rigor matemático superior que se traduce en filtros IIR digitales más exactos y un costo computacional acotado.
  \end{itemize}
\end{frame}
